1. **tests/test_BEHAV_01_init.sqf** — HQ Initialization test:

```sqf
// BEHAV-01: HQ Initialization Smoke Test
// Run via: execVM "tests\test_BEHAV_01_init.sqf" in debug console
// Prerequisites: Place HAL Core module synced to a group leader in editor
// Expected: After ~20s, HQ has personality traits and state variables set

private _testName = "BEHAV-01: HQ Init";
private _pass = 0;
private _fail = 0;
private _total = 0;

private _check = {
    params ["_desc", "_condition"];
    _total = _total + 1;
    if (_condition) then {
        _pass = _pass + 1;
        systemChat format ["  [PASS] %1", _desc];
    } else {
        _fail = _fail + 1;
        systemChat format ["  [FAIL] %1", _desc];
    };
};

systemChat format ["=== %1 ===", _testName];
systemChat "Waiting 20s for HQ init...";
sleep 20;

// Check leaderHQ exists
private _hq = missionNamespace getVariable ["leaderHQ", objNull];
["leaderHQ is not null", !(isNull _hq)] call _check;

if (isNull _hq) exitWith {
    systemChat format ["=== %1: ABORTED (no HQ) — %2/%3 passed ===", _testName, _pass, _total];
};

private _grp = group _hq;

// Check codeSign
private _code = _grp getVariable [QEGVAR(core,codeSign), ""];
["codeSign is set (expected 'A')", _code isEqualTo "A"] call _check;

// Check personality traits in range [0,1]
{
    private _val = _grp getVariable [_x select 0, -1];
    [format ["%1 in [0,1] (got %2)", _x select 1, _val], (_val >= 0) && (_val <= 1)] call _check;
} forEach [
    [QEGVAR(core,recklessness), "recklessness"],
    [QEGVAR(core,consistency), "consistency"],
    [QEGVAR(core,activity), "activity"],
    [QEGVAR(core,reflex), "reflex"],
    [QEGVAR(core,circumspection), "circumspection"],
    [QEGVAR(core,fineness), "fineness"]
];

// Check allHQ array
private _allHQ = missionNamespace getVariable [QEGVAR(core,allHQ), []];
["allHQ array is non-empty", count _allHQ > 0] call _check;

// Check allLeaders array
private _allLeaders = missionNamespace getVariable [QEGVAR(core,allLeaders), []];
["allLeaders array is non-empty", count _allLeaders > 0] call _check;

systemChat format ["=== %1: %2/%3 passed, %4 failed ===", _testName, _pass, _total, _fail];
```
